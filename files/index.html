Python Backend — Capture EDSA Frames
YouTube blocks direct canvas frame capture (CORS). Run this Python script locally to capture frames from the stream and push detections to this dashboard in real time.
1. Install dependencies
# Run in your terminal pip install yt-dlp opencv-python requests flask flask-cors
2. Save as backend.py and run it
# backend.py import cv2, requests, base64, time from flask import Flask, jsonify from flask_cors import CORS app = Flask(__name__) CORS(app) RF_KEY = "YOUR_ROBOFLOW_KEY" RF_MODEL = "ph-vehicles/1" YT_URL = "https://www.youtube.com/watch?v=jDS1VeU66PQ" def get_stream_url(): import yt_dlp opts = {'format': 'best[ext=mp4]', 'quiet': True} with yt_dlp.YoutubeDL(opts) as ydl: info = ydl.extract_info(YT_URL, download=False) return info['url'] def infer_frame(frame): _, buf = cv2.imencode('.jpg', frame) b64 = base64.b64encode(buf).decode() url = f"https://serverless.roboflow.com/{RF_MODEL}?api_key={RF_KEY}" r = requests.post(url, data=b64, headers={"Content-Type": "application/x-www-form-urlencoded"}) return r.json().get("predictions", []) if __name__ == "__main__": stream = get_stream_url() cap = cv2.VideoCapture(stream) while True: ret, frame = cap.read() if not ret: break preds = infer_frame(frame) for p in preds: # This prints to console — wire to dashboard via WebSocket print(p["class"], round(p["confidence"]*100), "%") time.sleep(5) # capture every 5 seconds
3. Push detections to this dashboard (from your Python script or browser console)
// Call this from your browser console OR wire it from Python via WebSocket: window.pushDetection({ vehicle: "Jeepney", confidence: 94, frame: 512 }) window.pushDetection({ vehicle: "Tricycle", confidence: 87, frame: 513 }) window.pushDetection({ vehicle: "Bus", confidence: 91, frame: 514 })
